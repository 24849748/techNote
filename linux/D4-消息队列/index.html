    【linux 学习笔记】消息队列

    消息队列、共享内存 和 信号量 被统称为 system-V IPC

    ## 查看系统当前的 ipc 对象
ipcs -q/m/s

## 输出
--------- 消息队列 -----------
键        msqid      拥有者  权限     已用字节数 消息      

------------ 共享内存段 --------------
键        shmid      拥有者  权限     字节     连接数  状态      

--------- 信号量数组 -----------
键        semid      拥有者  权限     nsems    

    ## 删除
ipcrm -q/m/s [id]
    • 消息队列提供了一种从一个进程向另一个进程发送一个数据块的方法。
    • 每个数据块都被认为含有一个类型，接收进程可以独立地接收含有不同类型的数据结构。
    • 可以通过发送消息来避免命名管道的同步和阻塞问题。

    与信号管道对比¶

    与信号对比： * 信号承载信息量少，消息队列可以承载大量自定义数据

    与管道对比： * 与命名管道类似 * 消息队列也可以独立于发送和接收进程而存在，在进程终止时，消息队列及其内容并不会被删除 * 管道只能承载无格式字节流，消息队列提供有格式的字节流 * 消息队列可以实现消息的随机查询，消息不一定要以先进先出的顺序接收，也可以按消息的类型接收

    消息队列使用¶

    函数原型 

    #include <sys/types.h>
#include <sys/ipc.h>
#include <sys/msg.h>

// 创建或打开消息队列，可创建的数量收系统限制
int msgget(key_t key, int msgflg);

// 发送消息
int msgsnd(int msqid, const void *msgp, size_t msgsz, int msgflg);
// 接收消息，与 FIFO 不同，可以指定取走某条消息
ssize_t msgrcv(int msqid, void *msgp, size_t msgsz, long msgtyp, int msgflg);

// 控制函数
int msgctl(int msqid, int cmd, struct msqid_ds *buf);

    消息传输结构体定义 

    typedef struct _msg_buf {
    long mtype;       // 消息类型
    char mtext[512];  // 消息内容
} msg_buf_t;

    • type 变量类型必须是 long
    • text 可以是任意类型，内核只关心其大小

    代码示例¶

    发送¶

    #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <fcntl.h>
#include <errno.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <sys/ipc.h>
#include <sys/shm.h>

#define MSG_MAX_LEN (512)

typedef struct
{
    long msg_type;
    char msg_ctx[MSG_MAX_LEN];
} msg_t;

int main(int argc, char *argv[])
{
    int qid;
    msg_t msg;

    qid = msgget((key_t)1234, (IPC_CREAT | 0666));
    if (qid == -1)
    {
        printf("msgget\n");
        exit(1);
    }
    printf("open queue %d\n", qid);

    while (1)
    {
        printf("Enter some message to send:");
        fgets(msg.msg_ctx, MSG_MAX_LEN, stdin);
        msg.msg_type = getpid();

        if (msgsnd(qid, &msg, strlen(msg.msg_ctx), 0) < 0)
        {
            printf("msgsnd error\n");
            exit(1);
        }
        else
        {
            printf("msgsnd ok\n");
        }

        if ((strncmp(msg.msg_ctx, "q", 1) == 0) 
            || (strncmp(msg.msg_ctx, "quit", 4) == 0))
        {
            printf("quit");
            break;
        }
    }

    return 0;
}

    接收¶

    #include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <unistd.h>
#include <string.h>

#define MSG_MAX_LEN (512)

typedef struct
{
    long msg_type;
    char msg_ctx[MSG_MAX_LEN];
} msg_t;

int main(int argc, char *argv[])
{
    int qid;
    msg_t msg;

    qid = msgget((key_t)1234, (IPC_CREAT | 0666));
    if (qid == -1)
    {
        printf("msgget\n");
        exit(1);
    }
    printf("open queue %d\n", qid);

    while (1)
    {
        /*读取消息队列*/
        memset(msg.msg_ctx, 0, MSG_MAX_LEN);

        if (msgrcv(qid, (void*)&msg, MSG_MAX_LEN, 0, 0) < 0)
        {
            printf("msgrcv error\n");
            exit(1);
        }

        printf("msgrcv[%ld]: %s", msg.msg_type, msg.msg_ctx);

        if ((strncmp(msg.msg_ctx, "q", 1) == 0) 
            || (strncmp(msg.msg_ctx, "quit", 4) == 0))
        {
            printf("quit");
            break;
        }
    }

    // 从系统内核删除消息队列
    if ((msgctl(qid, IPC_RMID, NULL)) < 0)
    {
        printf("msgctl error\n");
        exit(1);
    }
    else
    {
        printf("delete msg qid: %d\n", qid);
    }

    return 0;
}
